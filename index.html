  24. func ReadProfile(r io.Reader) ([]profile.Profile, error) {

  25.         files := make(map[string]*profile.Profile)

  26.         modeOk := false

  27.         id := 1

  28. 

  29.         s := bufio.NewScanner(r)

  30.         for s.Scan() {

  31.                 line := s.Text()

  32. 

  33.                 // first line must be "mode: xxx"

  34.                 if !modeOk {

  35.                         const p = "mode: "

  36.                         if !strings.HasPrefix(line, p) || line == p {

  37.                                 return nil, fmt.Errorf("first line must be mode: %q", line)

  38.                         }

  39.                         modeOk = true

  40.                         continue

  41.                 }

  42. 

  43.                 matches := reg.FindStringSubmatch(line)

  44.                 if matches == nil {

  45.                         return nil, fmt.Errorf("%q does not match expected format: %v", line, reg)

  46.                 }

  47. 

  48.                 fileName := matches[1]

  49.                 p := files[fileName]

  50. 

  51.                 if p == nil {

  52.                         p = &profile.Profile{ID: id, FileName: fileName}

  53.                         files[fileName] = p

  54.                         id++

  55.                 }

  56. 

  57.                 p.Blocks = append(p.Blocks, profile.Block{

  58.                         StartLine: toInt(matches[2]),

  59.                         StartCol:  toInt(matches[3]),

  60.                         EndLine:   toInt(matches[4]),

  61.                         EndCol:    toInt(matches[5]),

  62.                         NumState:  toInt(matches[6]),

  63.                         Count:     toInt(matches[7]),

  64.                 })

  65.         }

  66. 

  67.         return toProfiles(files)

  68. }

  78. func toProfiles(files map[string]*profile.Profile) ([]profile.Profile, error) {

  79.         dirs, err := importPath(files)

  80.         if err != nil {

  81.                 return nil, err

  82.         }

  83. 

  84.         profiles := make([]profile.Profile, 0, len(files))

  85.         for _, p := range files {

  86.                 p.Blocks = filterBlocks(p.Blocks)

  87.                 sort.SliceStable(p.Blocks, func(i, j int) bool {

  88.                         bi, bj := p.Blocks[i], p.Blocks[j]

  89.                         return bi.StartLine < bj.StartLine || bi.StartLine == bj.StartLine && bi.StartCol < bj.StartCol

  90.                 })

  91. 

  92.                 p.Dir = dirs[path.Dir(p.FileName)]

  93. 

  94.                 profiles = append(profiles, *p)

  95.         }

  96. 

  97.         sort.SliceStable(profiles, func(i, j int) bool {

  98.                 return profiles[i].FileName < profiles[j].FileName

  99.         })

  100. 

  101.         return profiles, nil

  102. }

  104. func filterBlocks(blocks []profile.Block) []profile.Block {

  105.         index := func(bs []profile.Block, b *profile.Block) int {

  106.                 for i, bb := range bs {

  107.                         if bb.StartLine == b.StartLine &&

  108.                                 bb.StartCol == b.StartCol &&

  109.                                 bb.EndLine == b.EndLine &&

  110.                                 bb.EndCol == b.EndCol {

  111.                                 return i

  112.                         }

  113.                 }

  114.                 return -1

  115.         }

  116. 

  117.         newBlocks := make([]profile.Block, 0, len(blocks))

  118.         for _, b := range blocks {

  119.                 i := index(newBlocks, &b)

  120.                 if i == -1 {

  121.                         newBlocks = append(newBlocks, b)

  122.                         continue

  123.                 }

  124.                 if b.Count > 0 {

  125.                         newBlocks[i] = b

  126.                 }

  127.         }

  128. 

  129.         return newBlocks

  130. }

  132. func importPath(files map[string]*profile.Profile) (map[string]string, error) {

  133.         dirs := make([]string, 0, len(files))

  134.         pkgs := make(map[string]string)

  135. 

  136.         for _, p := range files {

  137.                 if p.IsRelativeOrAbsolute() {

  138.                         continue

  139.                 }

  140. 

  141.                 dir := path.Dir(p.FileName)

  142.                 if _, ok := pkgs[dir]; !ok {

  143.                         pkgs[dir] = ""

  144.                         dirs = append(dirs, dir)

  145.                 }

  146.         }

  147. 

  148.         if len(dirs) == 0 {

  149.                 return pkgs, nil

  150.         }

  151. 

  152.         cmdName := filepath.Join(runtime.GOROOT(), "bin/go")

  153.         args := append([]string{"list", "-e", "-json"}, dirs...)

  154.         cmd := exec.Command(cmdName, args...)

  155. 

  156.         stdout, err := cmd.Output()

  157.         if err != nil {

  158.                 return nil, err

  159.         }

  160. 

  161.         type pkg struct {

  162.                 Dir        string

  163.                 ImportPath string

  164.                 Error      *struct {

  165.                         Err string

  166.                 }

  167.         }

  168. 

  169.         dec := json.NewDecoder(bytes.NewReader(stdout))

  170.         for {

  171.                 var p pkg

  172.                 err := dec.Decode(&p)

  173.                 if err == io.EOF {

  174.                         break

  175.                 }

  176.                 if err != nil {

  177.                         return nil, fmt.Errorf("decoding go list json: %v", err)

  178.                 }

  179.                 if p.Error != nil {

  180.                         return nil, fmt.Errorf(p.Error.Err)

  181.                 }

  182.                 pkgs[p.ImportPath] = p.Dir

  183.         }

  184. 

  185.         return pkgs, nil

  186. }

  95. func writeSource(buf *bytes.Buffer, src []byte, prof *profile.Profile) {

  96.         bi := 0

  97.         si := 0

  98.         line := 1

  99.         col := 1

  100.         cov0 := false

  101.         cov1 := false

  102. 

  103.         buf.WriteString("<ol>")

  104. 

  105.         for si < len(src) {

  106.                 if col == 1 {

  107.                         buf.WriteString("<li>")

  108.                         if cov0 {

  109.                                 buf.WriteString(`<span class="cov0">`)

  110.                         }

  111.                         if cov1 {

  112.                                 buf.WriteString(`<span class="cov1">`)

  113.                         }

  114.                 }

  115. 

  116.                 if len(prof.Blocks) > bi {

  117.                         block := prof.Blocks[bi]

  118.                         if block.StartLine == line && block.StartCol == col {

  119.                                 if block.Count == 0 {

  120.                                         buf.WriteString(`<span class="cov0">`)

  121.                                         cov0 = true

  122.                                 } else {

  123.                                         buf.WriteString(`<span class="cov1">`)

  124.                                         cov1 = true

  125.                                 }

  126.                         }

  127.                         if block.EndLine == line && block.EndCol == col || line > block.EndLine {

  128.                                 buf.WriteString(`</span>`)

  129.                                 bi++

  130.                                 cov0 = false

  131.                                 cov1 = false

  132.                                 continue

  133.                         }

  134.                 }

  135. 

  136.                 b := src[si]

  137.                 buf.WriteString(escape(b))

  138. 

  139.                 if b == '\n' {

  140.                         if cov0 || cov1 {

  141.                                 buf.WriteString("</span>")

  142.                         }

  143.                         buf.WriteString("</li>")

  144.                         line++

  145.                         col = 0

  146.                 }

  147. 

  148.                 si++

  149.                 col++

  150.         }

  151. 

  152.         buf.WriteString("</ol>")

  153. }

  18. func Create(profiles []profile.Profile) []Node {

  19.         nodes := make([]Node, 0)

  20.         for _, p := range profiles {

  21.                 addNode(&nodes, strings.Split(p.FileName, "/"), &p)

  22.         }

  23. 

  24.         return mergeSingreDir(nodes)

  27. func addNode(nodes *[]Node, paths []string, p *profile.Profile) {

  28.         name := paths[0]

  29.         nextPaths := paths[1:]

  30. 

  31.         idx := index(*nodes, name)

  32.         if idx == -1 {

  33.                 *nodes = append(*nodes, Node{

  34.                         Name:  name,

  35.                         Files: make([]profile.Profile, 0),

  36.                         Dirs:  make([]Node, 0),

  37.                 })

  38.                 idx = len(*nodes) - 1

  39.         }

  40. 

  41.         n := *nodes

  42.         if len(nextPaths) == 1 {

  43.                 n[idx].Files = append(n[idx].Files, *p)

  44.                 return

  45.         }

  46. 

  47.         addNode(&n[idx].Dirs, nextPaths, p)

  48. }

  50. func index(nodes []Node, name string) int {

  51.         for i, t := range nodes {

  52.                 if t.Name == name {

  53.                         return i

  54.                 }

  55.         }

  56.         return -1

  57. }

  59. func mergeSingreDir(nodes []Node) []Node {

  60.         for i, t := range nodes {

  61.                 mergeSingreDir(t.Dirs)

  62. 

  63.                 if len(t.Files) == 0 && len(t.Dirs) == 1 {

  64.                         sub := t.Dirs[0]

  65.                         nodes[i].Name = filepath.Join(t.Name, sub.Name)

  66.                         nodes[i].Files = sub.Files

  67.                         nodes[i].Dirs = sub.Dirs

  68.                 }

  69.         }

  70. 

  71.         return nodes

  72. }

  28. // Coverage returns covered ratio for file

  29. func (prof *Profile) Coverage() float64 {

  30.         var total, covered int64

  31.         for _, b := range prof.Blocks {

  32.                 total += int64(b.NumState)

  33.                 if b.Count > 0 {

  34.                         covered += int64(b.NumState)

  35.                 }

  36.         }

  37. 

  38.         if total == 0 {

  39.                 return 0

  40.         }

  41. 

  42.         return math.Round((float64(covered)/float64(total)*100)*10) / 10

  43. }
