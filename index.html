  107. func writeSource(buf *bytes.Buffer, src []byte, prof *profile.Profile) {

  109.         si := 0

  110.         line := 1

  111.         col := 1

  112.         cov0 := false

  113.         cov1 := false

  114. 

  115.         buf.WriteString("<ol>")

  117.         for si < len(src) {

  119.                         buf.WriteString("<li>")

  121.                                 buf.WriteString(`<span class="cov0">`)

  122.                         }

  123.                         if cov1 {

  124.                                 buf.WriteString(`<span class="cov1">`)

  125.                         }

  126.                 }

  127. 

  128.                 if len(prof.Blocks) > bi {

  129.                         block := prof.Blocks[bi]

  131.                                 if block.Count == 0 {

  132.                                         buf.WriteString(`<span class="cov0">`)

  133.                                         cov0 = true

  134.                                 } else {

  135.                                         buf.WriteString(`<span class="cov1">`)

  136.                                         cov1 = true

  137.                                 }

  138.                         }

  139.                         if block.EndLine == line && block.EndCol == col || line > block.EndLine {

  140.                                 buf.WriteString(`</span>`)

  141.                                 bi++

  142.                                 cov0 = false

  143.                                 cov1 = false

  144.                                 continue

  145.                         }

  146.                 }

  147. 

  148.                 b := src[si]

  149.                 switch b {

  150.                 case '<':

  151.                         buf.WriteString("&lt;")

  152.                 case '>':

  153.                         buf.WriteString("&gt;")

  154.                 case '&':

  155.                         buf.WriteString("&amp;")

  156.                 case '\t':

  157.                         buf.WriteString("        ")

  158.                 default:

  159.                         buf.WriteByte(b)

  160.                 }

  161. 

  162.                 if b == '\n' {

  163.                         if cov0 || cov1 {

  164.                                 buf.WriteString("</span>")

  165.                         }

  166.                         buf.WriteString("</li>")

  167.                         line++

  168.                         col = 0

  169.                 }

  170. 

  171.                 si++

  172.                 col++

  173.         }

  174. 

  175.         buf.WriteString("</ol>")

  23. 

  24.                 if b.Count > 0 {

  32. func (blocks *Blocks) Coverage() float64 {

  33.         var total, covered int64

  34.         for _, b := range *blocks {

  35.                 total += int64(b.NumStmt)

  36.                 if b.Count > 0 {

  37.                         covered += int64(b.NumStmt)

  38.                 }

  39.         }

  40. 

  41.         if total == 0 {

  42.                 return 0

  43.         }

  44. 

  45.         return float64(covered) / float64(total) * 100

  46. }

  14. func Scan(s *bufio.Scanner) (Profiles, error) {

  15.         files := make(map[string]*Profile)

  16.         mode := ""

  17. 

  18.         for s.Scan() {

  19.                 line := s.Text()

  20. 

  21.                 if mode == "" {

  22.                         const p = "mode: "

  23.                         if !strings.HasPrefix(line, p) || line == p {

  24.                                 return nil, fmt.Errorf("bad mode line: %q", line)

  25.                         }

  26.                         mode = line[len(p):]

  27.                         continue

  28.                 }

  29. 

  30.                 m := reg.FindStringSubmatch(line)

  31.                 if m == nil {

  32.                         return nil, fmt.Errorf("line %q does not match expected format: %v", line, reg)

  33.                 }

  34. 

  35.                 fileName := m[1]

  36.                 p := files[fileName]

  37. 

  38.                 if p == nil {

  39.                         p = &Profile{FileName: fileName, Mode: mode}

  40.                         files[fileName] = p

  41.                 }

  42. 

  43.                 p.Blocks = append(p.Blocks, Block{

  44.                         StartLine: toInt(m[2]),

  45.                         StartCol:  toInt(m[3]),

  46.                         EndLine:   toInt(m[4]),

  47.                         EndCol:    toInt(m[5]),

  48.                         NumStmt:   toInt(m[6]),

  49.                         Count:     toInt(m[7]),

  50.                 })

  51.         }

  52. 

  53.         return toProfiles(files), nil

  54. }

  56. func toProfiles(files map[string]*Profile) Profiles {

  57.         profiles := make([]Profile, 0, len(files))

  58.         for _, p := range files {

  59.                 profiles = append(profiles, *p)

  60.         }

  61. 

  62.         sort.SliceStable(profiles, func(i, j int) bool {

  63.                 return profiles[i].FileName < profiles[j].FileName

  64.         })

  65. 

  66.         for i, p := range profiles {

  67.                 profiles[i].ID = i

  68. 

  69.                 blocks := p.Blocks.Filter()

  70.                 sort.SliceStable(blocks, func(i, j int) bool {

  71.                         bi, bj := blocks[i], blocks[j]

  72.                         return bi.StartLine < bj.StartLine || bi.StartLine == bj.StartLine && bi.StartCol < bj.StartCol

  73.                 })

  74.                 profiles[i].Blocks = blocks

  75.         }

  76. 

  77.         return profiles

  78. }

  22.                         Name:     name,

  31.                 t[idx].Profiles = append(t[idx].Profiles, *p)

  35. }
