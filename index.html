  105. func genSource(buf *bytes.Buffer, src []byte, prof *profile.Profile) {

  106.         bi := 0

  107.         si := 0

  108.         line := 1

  109.         col := 1

  110.         cov0 := false

  111.         cov1 := false

  112. 

  113.         buf.WriteString("<ol>")

  114. 

  115.         for si < len(src) {

  116.                 if col == 1 {

  117.                         buf.WriteString("<li>")

  118.                         if cov0 {

  119.                                 buf.WriteString(`<span class="cov0">`)

  120.                         }

  121.                         if cov1 {

  122.                                 buf.WriteString(`<span class="cov1">`)

  123.                         }

  124.                 }

  125. 

  126.                 if len(prof.Blocks) > bi {

  127.                         block := prof.Blocks[bi]

  128.                         if block.StartLine == line && block.StartCol == col {

  129.                                 if block.Count == 0 {

  130.                                         buf.WriteString(`<span class="cov0">`)

  131.                                         cov0 = true

  132.                                 } else {

  133.                                         buf.WriteString(`<span class="cov1">`)

  134.                                         cov1 = true

  135.                                 }

  136.                         }

  137.                         if block.EndLine == line && block.EndCol == col || line > block.EndLine {

  138.                                 buf.WriteString(`</span>`)

  139.                                 bi++

  140.                                 cov0 = false

  141.                                 cov1 = false

  142.                                 continue

  143.                         }

  144.                 }

  145. 

  146.                 b := src[si]

  147.                 switch b {

  148.                 case '<':

  149.                         buf.WriteString("&lt;")

  150.                 case '>':

  151.                         buf.WriteString("&gt;")

  152.                 case '&':

  153.                         buf.WriteString("&amp;")

  154.                 case '\t':

  155.                         buf.WriteString("        ")

  156.                 default:

  157.                         buf.WriteByte(b)

  158.                 }

  159. 

  160.                 if b == '\n' {

  161.                         if cov0 || cov1 {

  162.                                 buf.WriteString("</span>")

  163.                         }

  164.                         buf.WriteString("</li>")

  165.                         line++

  166.                         col = 0

  167.                 }

  168. 

  169.                 si++

  170.                 col++

  171.         }

  172. 

  173.         buf.WriteString("</ol>")

  174. }

  14. func (blocks *Blocks) Coverage() float64 {

  15.         var total, covered int64

  16.         for _, b := range *blocks {

  17.                 total += int64(b.NumStmt)

  18.                 if b.Count > 0 {

  19.                         covered += int64(b.NumStmt)

  20.                 }

  21.         }

  22. 

  23.         if total == 0 {

  24.                 return 0

  25.         }

  26. 

  27.         return float64(covered) / float64(total) * 100

  28. }

  23. func Scan(s *bufio.Scanner) (Profiles, error) {

  24.         files := make(map[string]*Profile)

  25.         mode := ""

  26.         id := 1

  27. 

  28.         for s.Scan() {

  29.                 line := s.Text()

  30. 

  31.                 if mode == "" {

  32.                         const p = "mode: "

  33.                         if !strings.HasPrefix(line, p) || line == p {

  34.                                 return nil, fmt.Errorf("bad mode line: %q", line)

  35.                         }

  36.                         mode = line[len(p):]

  37.                         continue

  38.                 }

  39. 

  40.                 m := reg.FindStringSubmatch(line)

  41.                 if m == nil {

  42.                         return nil, fmt.Errorf("line %q does not match expected format: %v", line, reg)

  43.                 }

  44. 

  45.                 fileName := m[1]

  46.                 p := files[fileName]

  47. 

  48.                 if p == nil {

  49.                         p = &Profile{

  50.                                 FileName: fileName,

  51.                                 Mode:     mode,

  52.                         }

  53.                         files[fileName] = p

  54.                         id++

  55.                 }

  56. 

  57.                 p.Blocks = append(p.Blocks, Block{

  58.                         StartLine: toInt(m[2]),

  59.                         StartCol:  toInt(m[3]),

  60.                         EndLine:   toInt(m[4]),

  61.                         EndCol:    toInt(m[5]),

  62.                         NumStmt:   toInt(m[6]),

  63.                         Count:     toInt(m[7]),

  64.                 })

  65.         }

  66. 

  67.         return toProfiles(files), nil

  68. }

  70. func toProfiles(files map[string]*Profile) Profiles {

  71.         profiles := make([]Profile, 0, len(files))

  72.         for _, p := range files {

  73.                 p.Blocks = filterBlocks(p.Blocks)

  74.                 profiles = append(profiles, *p)

  75.         }

  76. 

  77.         sort.SliceStable(profiles, func(i, j int) bool {

  78.                 return profiles[i].FileName < profiles[j].FileName

  79.         })

  80. 

  81.         // ID振る

  82.         for i := range profiles {

  83.                 profiles[i].ID = i

  84.         }

  85. 

  86.         return profiles

  87. }

  89. func filterBlocks(blocks Blocks) Blocks {

  90.         pbm := make(map[string]Block)

  91.         for _, b := range blocks {

  92.                 // TODO: やり方考える

  93.                 k := fmt.Sprintf("%d-%d-%d-%d", b.StartLine, b.StartCol, b.EndLine, b.EndCol)

  94.                 if _, ok := pbm[k]; !ok {

  95.                         pbm[k] = b

  96.                 }

  97.                 if b.Count > 0 {

  98.                         pbm[k] = b

  99.                 }

  100.         }

  101. 

  102.         pbs := make(Blocks, 0, len(pbm))

  103.         for _, b := range pbm {

  104.                 pbs = append(pbs, b)

  105.         }

  106. 

  107.         sort.SliceStable(pbs, func(i, j int) bool {

  108.                 bi, bj := pbs[i], pbs[j]

  109.                 return bi.StartLine < bj.StartLine || bi.StartLine == bj.StartLine && bi.StartCol < bj.StartCol

  110.         })

  111. 

  112.         return pbs

  113. }

  27. func (tree *Tree) add(paths []string, p *Profile) {

  28.         name := paths[0]

  29.         nextPaths := paths[1:]

  30. 

  31.         idx, ok := tree.findNode(name)

  32.         if !ok {

  33.                 *tree = append(*tree, Node{

  34.                         Name:     name,

  35.                         Profiles: make(Profiles, 0),

  36.                         SubDirs:  make(Tree, 0),

  37.                 })

  38.                 idx = len(*tree) - 1

  39.         }

  40. 

  41.         t := *tree

  42.         if len(nextPaths) == 1 {

  43.                 t[idx].Profiles = append(t[idx].Profiles, *p)

  44.                 return

  45.         }

  46.         t[idx].SubDirs.add(nextPaths, p)

  47. }

  49. func (tree *Tree) findNode(name string) (int, bool) {

  50.         for i, t := range *tree {

  51.                 if t.Name == name {

  52.                         return i, true

  53.                 }

  54.         }

  55.         return 0, false

  56. }

  59. func (tree *Tree) merge() {

  60.         tt := *tree

  61.         for i, t := range *tree {

  62.                 t.SubDirs.merge()

  68.                         tt[i].SubDirs = sub.SubDirs

  70.         }

  28. func (r *reader) Read(path string) ([]byte, error) {

  29.         file, err := r.find(path)

  30.         if err != nil {

  31.                 return nil, err

  33.

  38. func (r *reader) find(path string) (string, error) {

  39.         dir, file := filepath.Split(path)

  40. 

  41.         if r.pkg == nil {

  42.                 pkg, err := build.Import(dir, ".", build.FindOnly)

  43.                 if err != nil {

  44.                         return "", fmt.Errorf("can't find %q: %v", path, err)

  45.                 }

  46.                 r.pkg = pkg

  47. 

  48.                 src := filepath.Join(r.pkg.SrcRoot, path)

  49.                 if _, err := os.Stat(src); err == nil {

  50.                         r.hasSrcRoot = true

  51.                         return src, nil

  52.                 }

  53. 

  54.                 src = filepath.Join(r.pkg.Dir, file)

  55.                 if _, err := os.Stat(src); err == nil {

  56.                         r.hasSrcRoot = false

  57.                         r.prefix = strings.TrimSuffix(r.pkg.ImportPath, strings.TrimPrefix(r.pkg.Dir, r.pkg.Root))

  58.                         return src, nil

  59.                 }

  60.         }

  61. 

  62.         if r.hasSrcRoot {

  63.                 return filepath.Join(r.pkg.SrcRoot, path), nil

  64.         }

  65.         return filepath.Join(r.pkg.Root, strings.TrimPrefix(path, r.prefix)), nil
